// ─── Constants ────────────────────────────────────────────────────────────────

const ANTHROPIC = 'https://api.anthropic.com/v1/messages';

const RUNWAY_BASE = 'https://api.dev.runwayml.com/v1';

const DEFAULT_STYLE = 'Photographic, cinematic, ARRI camera aesthetic, professional TV commercial. Realistic lighting, shallow depth of field, warm color grade, high production value.';


// ─── State ────────────────────────────────────────────────────────────────────

let state = {

  title: '',

  script: '',

  frames: Array(8).fill(null).map(() => ({ visual: '', dialogue: '' })),

  sketchPrompts: Array(8).fill(null),

  sketches: Array(8).fill(null),

  photos: Array(8).fill(null),

  audioDuration: null,

  useProxy: false

};


// ─── Script parser ────────────────────────────────────────────────────────────

function parseScript(script) {

  const lines = script.split('\n').map(l => l.trim()).filter(Boolean);

  const isD = l => /^[A-Z®©\s]{2,}:\s+\S/.test(l) || /^(AVO|VO|V\.O\.|A\.V\.O\.):/i.test(l);

  const segs = []; let vis = [], dia = [];

  for (const line of lines) {

    if (isD(line)) dia.push(line); else vis.push(line);

    if (vis.length + dia.length >= 2 && segs.length < 7) {

      segs.push({ visual: vis.join(' '), dialogue: dia.join(' ') });

      vis = []; dia = [];

    }

  }

  if (vis.length + dia.length) segs.push({ visual: vis.join(' '), dialogue: dia.join(' ') });

  while (segs.length < 8) segs.push({ visual: '', dialogue: '' });

  return segs.slice(0, 8);

}


// ─── Claude API ───────────────────────────────────────────────────────────────

async function claude(prompt, maxTokens = 1200) {

  const res = await fetch(ANTHROPIC, {

    method: 'POST',

    headers: { 'Content-Type': 'application/json' },

    body: JSON.stringify({

      model: 'claude-sonnet-4-20250514',

      max_tokens: maxTokens,

      messages: [{ role: 'user', content: prompt }]

    })

  });

  const data = await res.json();

  return data.content?.[0]?.text?.trim() || '';

}


// ─── Generate 8 sketch prompts ────────────────────────────────────────────────

async function generateSketchPrompts(script, frames) {

  const segs = frames.map((f, i) =>

    `Frame ${i+1}: ${f.visual || '(no visual direction)'}${f.dialogue ? ' | Dialogue: ' + f.dialogue : ''}`

  ).join('\n');

  const raw = await claude(

    `You are a professional storyboard artist and prompt engineer. Given this TV commercial script and its 8 parsed scene segments, write exactly 8 image generation prompts — one per frame — for pencil-style storyboard sketches.\n\nFULL SCRIPT:\n${script}\n\nPARSED SEGMENTS:\n${segs}\n\nRules:\n- Each prompt must describe the scene composition visually: camera angle, subjects, setting, action\n- End every prompt with: "storyboard sketch, pencil illustration, black and white, loose gestural lines, no color, no text, no watermarks, 16:9 composition"\n- Be specific about what is in frame — characters, props, environment\n- Do not include dialogue or script text in the prompt\n- Return ONLY a JSON array of exactly 8 strings, no preamble, no markdown fences, no explanation`

  );

  try {

    const parsed = JSON.parse(raw.replace(/```json|```/g, '').trim());

    if (Array.isArray(parsed) && parsed.length === 8) return parsed;

  } catch {}

  return frames.map(f =>

    `${f.visual || 'cinematic establishing shot'}. storyboard sketch, pencil illustration, black and white, loose gestural lines, no color, no text, no watermarks, 16:9 composition`

  );

}


// ─── Generate single frame prompt ─────────────────────────────────────────────

async function generateFramePrompt(frame) {

  return await claude(

    `Write a single storyboard sketch image generation prompt for this scene.\nScene: ${frame.visual || 'cinematic establishing shot'}\nDialogue: ${frame.dialogue || 'none'}\nReturn ONLY the prompt string. End with: "storyboard sketch, pencil illustration, black and white, loose gestural lines, no color, no text, no watermarks, 16:9 composition"`,

    200

  );

}


// ─── Generate photo prompt ────────────────────────────────────────────────────

async function generatePhotoPrompt(frame) {

  const style = document.getElementById('style-input').value.trim() || DEFAULT_STYLE;

  return await claude(

    `You are a professional cinematographer and AI image prompt engineer. Write a single high-fidelity photographic image generation prompt for a TV commercial frame.\n\nSCENE: ${frame.visual || 'cinematic establishing shot'}\nDIALOGUE: ${frame.dialogue || 'none'}\nVISUAL STYLE: ${style}\n\nRules:\n- Apply the visual style to define lighting, color grade, lens, mood\n- Be specific and cinematic — describe what the camera sees\n- Do NOT include text, watermarks, or dialogue\n- Return ONLY the prompt string, no preamble, no quotes`,

    400

  );

}


// ─── Pollinations fetch → base64 ──────────────────────────────────────────────

async function fetchSketch(prompt) {

  const encoded = encodeURIComponent(prompt);

  const seed = Math.floor(Math.random() * 999999);

  const url = `https://image.pollinations.ai/prompt/${encoded}?width=1280&height=720&seed=${seed}&nologo=true&model=flux`;

  const res = await fetch(url);

  if (!res.ok) throw new Error(`Pollinations ${res.status}`);

  const blob = await res.blob();

  return new Promise((resolve, reject) => {

    const reader = new FileReader();

    reader.onload = () => resolve(reader.result);

    reader.onerror = reject;

    reader.readAsDataURL(blob);

  });

}


// ─── Runway generate → base64 ─────────────────────────────────────────────────

async function runwayGenerate(prompt) {

  const key = document.getElementById('runway-key').value.trim();

  const proxy = document.getElementById('proxy-url').value.trim();

  const useProxy = state.useProxy;

  const endpoint = useProxy ? `${proxy}/text_to_image` : `${RUNWAY_BASE}/text_to_image`;

  const headers = { 'Content-Type': 'application/json', 'X-Runway-Version': '2024-11-06' };

  if (!useProxy) headers['Authorization'] = `Bearer ${key}`;

  const res = await fetch(endpoint, {

    method: 'POST', headers,

    body: JSON.stringify({ model: 'gen4_image_turbo', promptText: prompt, ratio: '1280:720' })

  });

  if (!res.ok) throw new Error(`Runway ${res.status}`);

  const { id } = await res.json();

  for (let i = 0; i < 60; i++) {

    await new Promise(r => setTimeout(r, 2000));

    const pollUrl = useProxy ? `${proxy}/tasks/${id}` : `${RUNWAY_BASE}/tasks/${id}`;

    const poll = await fetch(pollUrl, { headers });

    const task = await poll.json();

    if (task.status === 'SUCCEEDED') {

      const imgUrl = task.output?.[0];

      if (!imgUrl) throw new Error('No output from Runway');

      const r2 = await fetch(imgUrl);

      const blob = await r2.blob();

      return new Promise((resolve, reject) => {

        const reader = new FileReader();

        reader.onload = () => resolve(reader.result);

        reader.onerror = reject;

        reader.readAsDataURL(blob);

      });

    }

    if (task.status === 'FAILED') throw new Error(task.failure || 'Runway failed');

  }

  throw new Error('Runway timed out');

}

function setStatus(msg) {

  const pill = document.getElementById('status-pill');

  else { pill.classList.remove('active'); txt.textContent = ''; }

}


function getCard(idx) { return document.querySelector(`.frame-card[data-idx="${idx}"]`); }


function setFrameLoading(idx, type) {

  // type: 'sketch' | 'photo' | null

  const card = getCard(idx);

  if (!card) return;

  const loading = card.querySelector('.frame-loading');

  const label = card.querySelector('.frame-loading-label');

  const spinner = loading?.querySelector('.spinner');

  if (type === 'sketch') {

    loading.style.display = 'flex';

    label.textContent = 'Sketching…';

    if (spinner) { spinner.classList.remove('green'); }

    card.querySelector('.frame-img').style.display = 'none';

    card.querySelector('.frame-placeholder').style.display = 'none';

  } else if (type === 'photo') {

    loading.style.display = 'flex';

    label.textContent = 'Generating photo…';

    if (spinner) spinner.classList.add('green');

  } else {

    loading.style.display = 'none';

  }

}


function setFrameImage(idx, dataUrl, isPhoto) {

  const card = getCard(idx);

  if (!card) return;

  const img = card.querySelector('.frame-img');

  const photoBadge = card.querySelector('.photo-badge');

  img.src = dataUrl;

  img.style.display = 'block';

  card.querySelector('.frame-placeholder').style.display = 'none';

  card.querySelector('.frame-loading').style.display = 'none';

  // Enable save button

  const saveBtn = card.querySelector('.btn-save');

  saveBtn.classList.remove('disabled');

  saveBtn.disabled = false;

}

// ─── Sketch ───────────────────────────────────────────────────────────────────

async function doSketch(idx, promptOverride) {

  clearFrameError(idx);

  setFrameLoading(idx, 'sketch');

  setPhotoBtn(idx, false);

  try {

    let prompt = promptOverride;

    if (!prompt) {

      setStatus(`Claude is writing a prompt for frame ${idx+1}…`);

      prompt = await generateFramePrompt(state.frames[idx]);

    }

    state.sketchPrompts[idx] = prompt;

    setFramePrompt(idx, prompt);

    setStatus(`Fetching sketch for frame ${idx+1}…`);

    const dataUrl = await fetchSketch(prompt);

    state.sketches[idx] = dataUrl;

    setFrameImage(idx, dataUrl, false);

    setPhotoBtn(idx, true);

  } catch(e) {

    setFrameError(idx, e.message + ' — click ↻ Sketch to retry');

    setFrameLoading(idx, null);

  }

  setStatus('');

}


// ─── Photo upscale ────────────────────────────────────────────────────────────

async function doUpscale(idx) {

  clearFrameError(idx);

  setFrameLoading(idx, 'photo');

  try {

    setStatus(`Claude is writing a photo prompt for frame ${idx+1}…`);

    const prompt = await generatePhotoPrompt(state.frames[idx]);

    setStatus(`Sending frame ${idx+1} to Runway Gen-4…`);

    const dataUrl = await runwayGenerate(prompt);

    state.photos[idx] = dataUrl;

    setFrameImage(idx, dataUrl, true);

  } catch(e) {

    setFrameError(idx, e.message);

    setFrameLoading(idx, null);

    // Restore sketch if available

    if (state.sketches[idx]) setFrameImage(idx, state.sketches[idx], false);

  }

  setStatus('');

}


// ─── Save frame ───────────────────────────────────────────────────────────────

function doSave(idx) {

  const dataUrl = state.photos[idx] || state.sketches[idx];

  if (!dataUrl) return;

  const a = document.createElement('a');

  a.href = dataUrl;

  a.download = `frame_${idx+1}.png`;

  a.click();

}


// ─── Full generate ────────────────────────────────────────────────────────────

async function handleGenerate() {

  const script = document.getElementById('script-input').value.trim();

  if (!script) return;

  state.title = document.getElementById('title-input').value.trim();

  state.script = script;

  state.frames = parseScript(script);

  state.sketches = Array(8).fill(null);

  state.photos = Array(8).fill(null);

  state.sketchPrompts = Array(8).fill(null);


  document.getElementById('board-title').textContent = state.title || 'Untitled Storyboard';

  document.getElementById('input-phase').style.display = 'none';

  document.getElementById('board-phase').style.display = 'block';

  document.getElementById('new-script-btn').style.display = 'inline-block';


  buildGrid();


  try {

    setStatus('Claude is reading the full script and writing 8 image prompts…');

    const prompts = await generateSketchPrompts(script, state.frames);

    state.sketchPrompts = prompts;

    setStatus('');

    for (let i = 0; i < 8; i++) {

      await doSketch(i, prompts[i]);

    }

  } catch(e) {

    setStatus('');

    for (let i = 0; i < 8; i++) {

      await doSketch(i, null);

    }

  }

  setStatus('');

}


// ─── Audio ────────────────────────────────────────────────────────────────────

function handleAudio(input) {

  const file = input.files[0]; if (!file) return;

  const url = URL.createObjectURL(file);

  const audio = new Audio(url);

  audio.onloadedmetadata = () => {

    state.audioDuration = audio.duration;

    const label = document.getElementById('audio-label');

    label.classList.add('has-file');

    document.getElementById('audio-text').textContent = `${file.name} · ${audio.duration.toFixed(1)}s`;

    document.getElementById('jsx-btn').disabled = false;

  };

}

// ─── Copy Worker Script ───────────────────────────────────────────────────────

function copyWorkerScript() {

  const code = `export default {

  async fetch(request, env) {

    const cors = {

      'Access-Control-Allow-Origin': '*',

      'Access-Control-Allow-Methods': 'GET,POST,OPTIONS',

      'Access-Control-Allow-Headers': 'Content-Type,X-Runway-Version',

    };

    if (request.method === 'OPTIONS') return new Response(null, { headers: cors });

    const url = new URL(request.url);

    const runwayUrl = 'https://api.dev.runwayml.com/v1' + url.pathname;

    const body = request.method !== 'GET' ? await request.text() : undefined;

    const res = await fetch(runwayUrl, {

      method: request.method,

      headers: {

        'Content-Type': 'application/json',

        'X-Runway-Version': '2024-11-06',

        'Authorization': \`Bearer \${env.RUNWAY_API_KEY}\`,

      },

      body,

    });

    return new Response(await res.text(), {

      status: res.status,

      headers: { ...cors, 'Content-Type': 'application/json' },

    });

  }

};`;

  navigator.clipboard.writeText(code).then(() => alert('✅ Worker script copied to clipboard!'));

}
